## Repositories {#repositories}
### Releases {#releases}
Use this repository to work with the release version of the API (recommended)

````mdx-code-block
<Tabs
  groupId="repositories"
  defaultValue="maven"
  values={[
  {label: "Maven", value: "maven"},
  {label: "Gradle", value: "gradle-groovy"},
  {label: "Gradle Kotlin DSL", value: "gradle-kotlin"}
]}>
  <TabItem value="maven">

  ```xml
  <repository>
      <id>cloudnet-releases</id>
      <url>https://repo.cloudnetservice.eu/repository/releases/</url>
  </repository>
  ```

  </TabItem>
  <TabItem value="gradle-groovy">

  ```groovy
  maven { url 'https://repo.cloudnetservice.eu/repository/releases/' }
  ```

  </TabItem>
  <TabItem value="gradle-kotlin">

  ```kotlin
  maven("https://repo.cloudnetservice.eu/repository/releases/")
  ```

  </TabItem>
  </Tabs>
````

## Dependencies {#dependencies}
You have to replace `${VERSION}` with the version which you want to use. (e.g. `3.3.0-RELEASE` or `3.4.0-SNAPSHOT`)

### cloudnet-driver {#cloudnet-driver}
This dependency includes the CloudNet-Driver API which can be used in plugins and modules.
````mdx-code-block
<Tabs
  groupId="dependencies"
  defaultValue="maven"
  values={[{label: "Maven", value: "maven"},
  {label: "Gradle", value: "gradle-groovy"},
  {label: "Gradle Kotlin DSL", value: "gradle-kotlin"},
  {label: "SBT", value: "sbt"},
  {label: "Ivy", value: "ivy"},
  {label: "Grape", value: "grape"},
  {label: "Leiningen", value: "leiningen"},
  {label: "Buildr", value: "buildr"}
]} >

  <TabItem value="maven">

  ```xml
  <cloudnet-driver>
      <groupId>de.dytanic.cloudnet</groupId>
      <artifactId>cloudnet-driver</artifactId>
      <version>${VERSION}</version>
      <scope>provided</scope>
  </cloudnet-driver>
  ```

  </TabItem>
  <TabItem value="gradle-groovy">

  ```groovy
  implementation 'de.dytanic.cloudnet:cloudnet-driver:${VERSION}'
  ```
  </TabItem>
  <TabItem value="gradle-kotlin">

  ```kotlin
  implementation("de.dytanic.cloudnet:cloudnet-driver:${VERSION}")
  ```

  </TabItem>
  <TabItem value="sbt">

  ```
  libraryDependencies += "de.dytanic.cloudnet" % "cloudnet-driver" % "${VERSION}"
  ```

  </TabItem>
  <TabItem value="ivy">

  ```xml
  <dependency org="de.dytanic.cloudnet" name="cloudnet-driver" rev="${VERSION}"/>
  ```

  </TabItem>
  <TabItem value="grape">

  ```
  @Grapes(
      @Grab(group='de.dytanic.cloudnet', module='cloudnet-driver', version='${VERSION}')
  )
  ```

  </TabItem>
  <TabItem value="leiningen">

  ```
  [de.dytanic.cloudnet/cloudnet-driver "${VERSION}"]
  ```

  </TabItem>
  <TabItem value="buildr">

  ```
  'de.dytanic.cloudnet:cloudnet-driver:jar:${VERSION}'
  ```

  </TabItem>
</Tabs>
````
