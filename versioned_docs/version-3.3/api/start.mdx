## Repositories
### Releases
Use this repository to work with the release version of the API (recommended)

<Tabs
  groupId="repositories"
  defaultValue="maven"
  values={[
  {label: "Maven", value: "maven"},
  {label: "Gradle", value: "gradle-groovy"},
  {label: "Gradle Kotlin DSL", value: "gradle-kotlin"}
]}>
  <TabItem value="maven">

  ```xml
  <repository>
      <id>cloudnet-releases</id>
      <url>https://repo.cloudnetservice.eu/repository/releases/</url>
  </repository>
  ```

  </TabItem>
  <TabItem value="gradle-groovy">

  ```groovy
  maven { url 'https://repo.cloudnetservice.eu/repository/releases/' }
  ```

  </TabItem>
  <TabItem value="gradle-kotlin">

  ```kotlin
  maven("https://repo.cloudnetservice.eu/repository/releases/")
  ```

  </TabItem>
  </Tabs>

## Dependencies
You have to replace `${VERSION}` with the version which you want to use. (e.g. `3.3.0-RELEASE` or `3.4.0-SNAPSHOT`)

### cloudnet
CloudNet application for modules (NOT AVAILABLE FOR PLUGINS!)

  groupId="dependencies"
  defaultValue="maven"
  values={[{label: "Maven", value: "maven"},
  {label: "Gradle", value: "gradle-groovy"},
  {label: "Gradle Kotlin DSL", value: "gradle-kotlin"},
  {label: "SBT", value: "sbt"},
  {label: "Ivy", value: "ivy"},
  {label: "Grape", value: "grape"},
  {label: "Leiningen", value: "leiningen"},
  {label: "Buildr", value: "buildr"}
]} >

  <TabItem value="maven">

  ```xml
  <dependency>
      <groupId>de.dytanic.cloudnet</groupId>
      <artifactId>cloudnet</artifactId>
      <version>${VERSION}</version>
      <scope>provided</scope>
  </dependency>
  ```

  </TabItem>
  <TabItem value="gradle-groovy">

  ```groovy
  implementation 'de.dytanic.cloudnet:cloudnet:${VERSION}'
  ```
  </TabItem>
  <TabItem value="gradle-kotlin">

  ```kotlin
  implementation("de.dytanic.cloudnet:cloudnet:${VERSION}")
  ```

  </TabItem>
  <TabItem value="sbt">

  ```
  libraryDependencies += "de.dytanic.cloudnet" % "cloudnet" % "${VERSION}"
  ```

  </TabItem>
  <TabItem value="ivy">

  ```xml
  <dependency org="de.dytanic.cloudnet" name="cloudnet" rev="${VERSION}"/>
  ```

  </TabItem>
  <TabItem value="grape">

  ```
  @Grapes(
      @Grab(group='de.dytanic.cloudnet', module='cloudnet', version='${VERSION}')
  )
  ```

  </TabItem>
  <TabItem value="leiningen">

  ```
  [de.dytanic.cloudnet/cloudnet "${VERSION}"]
  ```

  </TabItem>
  <TabItem value="buildr">

  ```
  'de.dytanic.cloudnet:cloudnet:jar:${VERSION}'
  ```

  </TabItem>
</Tabs>
